### AdvancedInteractiveMap Props

```typescript
interface AdvancedMapProps {
  properties: PropertyMarker[]        // Array of property locations
  selectedArea: string | null         // Currently selected area name
  onAreaSelect: (area: string) => void // Callback when area clicked
  height: string                      // Tailwind height class (h-96)
}

interface PropertyMarker {
  id: string                          // UUID
  location: string                    // Area name
  price: number                       // Total price
  priceSqft: number                   // Price per sqft
  coordinates: [number, number]       // [latitude, longitude]
  status: string                      // available|under_construction|sold|rented
}
```
